PHP 7 新特性

2016年4月3日
| PHP

1. 太空船运算符

echo 1<=>1; //0
echo 2<=>1; //1
echo 1<=>2; //-1

2. 类型声明

declare(strict_type=1)//strict_type=1表示开启严格模式
function sumOfInts(int ...$ints):int{
    return array_sum($ints);
}

3.null 合并操作符

$page = isset($_GET['page']) ? $_GET['page'] : 0;
$page = $_GET['page'] ?? 0;

4. 常量数组

define('ANIMALS',['dog','cat']);

5.namespace 批量导入

use Space\{ClassA,ClassB as B,ClassC}

6.intdiv 函数

intdiv(10,3);

7.list 方括号

$arr = [1,2,3];
list($a,$b,$c) = $arr;

$arr = [1,2,3];
[$a,$b,$c] = $arr;
